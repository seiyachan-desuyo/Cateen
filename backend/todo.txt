from flask import Flask, request, jsonify
from flask_cors import CORS
from openai import OpenAI
# 导入我们自己写的模块
from file import DataManager 

app = Flask(__name__, static_folder='../frontend', static_url_path='/')
CORS(app)

# 实例化数据管理器
data_manager = DataManager()

DEEPSEEK_API_KEY = 'sk-a29e901493ee46e290f20696adbfa4a3' 
client = OpenAI(api_key=DEEPSEEK_API_KEY, base_url='https://api.deepseek.com')

INIT_PROMPT = "嘿，京爷儿您来了，真地道,今儿吃点儿啥？..."

# 动态生成 System Prompt
def get_system_prompt():
    menu_str = data_manager.get_menu_as_string()
    return f'''
你是一个食堂推荐系统，请根据以下最新的菜单数据进行推荐：
{menu_str}

回复规则：
1. 必须基于上述数据推荐。
2. 如果用户没有提供足够信息（如辣度、价格），请追问。
3. 语气要亲切活泼。
'''

msg_history = [] 

@app.route('/')
def index():
    return app.send_static_file('index.html')

@app.route('/chat', methods=['POST'])
def chat():
    global msg_history
    
    # 如果是新对话，初始化System Prompt
    if not msg_history:
        msg_history.append({"role": "system", "content": get_system_prompt()}) 
    
    user_message = request.json.get('message', '')
    
    # 1. 记录用户输入到内存
    msg_history.append({"role": "user", "content": user_message })
    # 2. 记录用户输入到文件 (调用 file.py 的功能)
    data_manager.log_conversation("User", user_message)

    try:
        response = client.chat.completions.create(
            model="deepseek-chat",
            messages=msg_history,
            stream=False
        )
        
        ai_reply = response.choices[0].message.content
        
        # 3. 记录AI回复到内存
        msg_history.append({"role": "assistant", "content": ai_reply})
        # 4. 记录AI回复到文件
        data_manager.log_conversation("AI", ai_reply)

        return jsonify({"reply": ai_reply})
        
    except Exception as e:
        print(f"Error: {e}")
        return jsonify({"reply": "哎呀，贸小饭脑子卡壳了，请稍后再试～"}), 500

@app.route('/add_dish', methods=['POST'])
def api_add_dish():
    """
    这是一个隐藏接口，可以用 Postman 调用，
    或者你可以在前端写一个管理员页面来调用它。
    """
    data = request.json
    location = data.get('location')
    dish = data.get('dish')
    price = data.get('price')
    tags = data.get('tags')
    
    if location and dish and price:
        data_manager.add_dish(location, dish, price, tags)
        # 清空历史，迫使下次对话重新加载新菜单
        global msg_history
        msg_history = [] 
        return jsonify({"status": "success", "msg": f"已成功上架新菜品：{dish}"})
    else:
        return jsonify({"status": "error", "msg": "信息不全"}), 400

@app.route('/init', methods=['GET'])
def init():
    return jsonify({"reply": INIT_PROMPT})

if __name__ == '__main__':
    app.run(debug=True)

    if not data_manager.menu_data:

    )
    
    # 同步到历史记录，让AI知道刚才发生了什么

    return jsonify({"reply": reply_text})
